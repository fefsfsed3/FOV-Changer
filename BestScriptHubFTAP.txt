local Players = game:GetService("Players")
local player = Players.LocalPlayer
local replicatedStorage = game:GetService("ReplicatedStorage")
local debris = game:GetService("Debris")
local chatEvent = replicatedStorage:FindFirstChild("DefaultChatSystemChatEvents") and replicatedStorage.DefaultChatSystemChatEvents:FindFirstChild("SayMessageRequest")

task.spawn(function()
    if chatEvent then
        chatEvent:FireServer("I JUST GOT TROLLED BY TROLL SCRIPT 1.0.0 CUZ I TRIED LO0PING OR AIM LOCKING AND I GOT SNITCHED", "All")
    end
    task.wait(8)
    if chatEvent then
        chatEvent:FireServer("I GOT SNITCHED BY THIS TROLL SCRIPT!! I THOUGHT IT WAS A LO0P SCRIPT TO TROLL!", "All")
    end
end)

local crazyLag = false
local crazySpin = false

task.spawn(function()
    while task.wait(1) do
        local screenGui = Instance.new("ScreenGui", player.PlayerGui)
        local textLabel = Instance.new("TextLabel", screenGui)
        textLabel.Size = UDim2.new(0.3, 0, 0.1, 0)
        textLabel.Position = UDim2.new(math.random(), 0, math.random(), 0)
        textLabel.BackgroundColor3 = Color3.fromRGB(math.random(0,255), math.random(0,255), math.random(0,255))
        textLabel.Text = "GOT TROLLED!!"
        textLabel.TextColor3 = Color3.new(1,1,1)
        textLabel.TextScaled = true
        debris:AddItem(screenGui, 3)
        if crazyLag then
            for i = 1,5 do
                local moreGui = Instance.new("ScreenGui", player.PlayerGui)
                local moreLabel = Instance.new("TextLabel", moreGui)
                moreLabel.Size = UDim2.new(0.3, 0, 0.1, 0)
                moreLabel.Position = UDim2.new(math.random(), 0, math.random(), 0)
                moreLabel.BackgroundColor3 = Color3.fromRGB(math.random(0,255), math.random(0,255), math.random(0,255))
                moreLabel.Text = "GOT TROLLED HARDER!!"
                moreLabel.TextColor3 = Color3.new(1,1,1)
                moreLabel.TextScaled = true
                debris:AddItem(moreGui, 2)
            end
        end
    end
end)

task.spawn(function()
    while task.wait(0.03) do
        local cam = workspace.CurrentCamera
        if cam then
            if crazySpin then
                cam.CFrame = cam.CFrame * CFrame.Angles(0, math.rad(30), 0)
            else
                cam.CFrame = cam.CFrame * CFrame.Angles(0, math.rad(10), 0)
            end
        end
    end
end)

task.spawn(function()
    while task.wait(0.05) do
        local part = Instance.new("Part")
        part.Size = Vector3.new(2,2,2)
        part.Position = player.Character and player.Character:GetPivot().Position + Vector3.new(math.random(-10,10), math.random(2,5), math.random(-10,10)) or Vector3.new()
        part.Anchored = true
        part.CanCollide = false
        part.Transparency = 1
        part.Parent = workspace
        debris:AddItem(part, 3)
        if crazyLag then
            for i = 1,5 do
                local morePart = Instance.new("Part")
                morePart.Size = Vector3.new(2,2,2)
                morePart.Position = player.Character and player.Character:GetPivot().Position + Vector3.new(math.random(-30,30), math.random(5,10), math.random(-30,30)) or Vector3.new()
                morePart.Anchored = true
                morePart.CanCollide = false
                morePart.Transparency = 1
                morePart.Parent = workspace
                debris:AddItem(morePart, 2)
            end
        end
    end
end)
